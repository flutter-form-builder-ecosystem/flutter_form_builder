@JSName('createWriter')
Future<FileWriter> createWriter() {
  var completer = new Completer<FileWriter>();
  _createWriter((value) {
    applyExtension('FileWriter', value);
    completer.complete(value);
  }, (error) {
    completer.completeError(error);
  });
  return completer.future;
}
